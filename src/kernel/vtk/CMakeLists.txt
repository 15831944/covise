ADD_DEFINITIONS(-DCOVISE_VTK)

SET(VTK_SOURCES
    coVtk.cpp
)

SET(VTK_HEADERS
    coVtk.h
)

USING(VTK optional)


ADD_COVISE_LIBRARY(coVtk ${COVISE_LIB_TYPE} ${VTK_SOURCES} ${VTK_HEADERS})
TARGET_LINK_LIBRARIES(coVtk coAppl coApi coCore coConfig ${EXTRA_LIBS})

IF(NOT MSVC)
   ADD_COVISE_COMPILE_FLAGS(coVtk "-Wno-deprecated-declarations")
ENDIF()

COVISE_INSTALL_TARGET(coVtk)
COVISE_INSTALL_HEADERS(vtk ${VTK_HEADERS})
